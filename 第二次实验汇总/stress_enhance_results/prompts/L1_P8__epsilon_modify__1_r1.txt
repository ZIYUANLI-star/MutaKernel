    const float* __restrict__ B,

    const int THREADS_PER_COL = TILE_N / (BLOCK_SIZE / TILE_M);

    for (int i = 0; i < TILE_N / THREADS_PER_COL; i++) {
        accum[i] = 0.0f;
    }

    for (int k_tile = 0; k_tile < K; k_tile += TILE_K) {
        // Load A tile into shared memory (coalesced access)
        for (int load_idx = thread_id; load_idx < TILE_M * TILE_K; load_idx += BLOCK_SIZE) {
            int row = load_idx / TILE_K;
            int col = load_idx % TILE_K;

            int global_col = k_tile + col;

                As[row][col] = A[global_row * K + global_col];
            } else {
                As[row][col] = 0.0f;
            }
        }

            int row = load_idx / TILE_N;
