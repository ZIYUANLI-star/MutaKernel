#include <cuda_runtime.h>
#include <cfloat>
#include <cub/cub.cuh>

template<typename scalar_t>
__device__ __forceinline__ scalar_t warp_reduce_max(scalar_t val) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = max(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

template<typename scalar_t>
__global__ void max_reduction_dim1_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int dim1,
    const int dim2,
    const int stride0,
    const int stride1,
    const int stride2) {
    
    const int batch_idx = blockIdx.x;
    const int dim2_idx = blockIdx.y;
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    scalar_t thread_max = -FLT_MAX;
    
    // Process multiple elements per thread for better memory coalescing
    const int elements_per_thread = (dim1 + blockDim.x - 1) / blockDim.x;
    const int start_idx = tid * elements_per_thread;
    const int end_idx = min(start_idx + elements_per_thread, dim1);
    
    for (int i = start_idx; i < end_idx; i++) {
        int idx = batch_idx * stride0 + i * stride1 + dim2_idx * stride2;
        scalar_t val = input[idx];
        if (val > thread_max) {
            thread_max = val;
        }
    }
    
    // Warp-level reduction
    scalar_t warp_max = warp_reduce_max(thread_max);
    
    // First thread in each warp writes to shared memory
    __shared__ scalar_t warp_maxs[32];
    if (lane_id == 0) {
        warp_maxs[warp_id] = warp_max;
    }
    __syncthreads();
    
    // Final reduction across warps (only first warp does this)
    if (warp_id == 0) {
        scalar_t val = (lane_id < (blockDim.x + 31) / 32) ? warp_maxs[lane_id] : -FLT_MAX;
        scalar_t block_max = warp_reduce_max(val);

        }
