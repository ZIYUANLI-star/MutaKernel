#define TILE_M 128
#define TILE_N 128
#define TILE_K 8
#define BLOCK_SIZE 256

__global__ void matmul_optimized_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,

    // Block indices
    int block_row = blockIdx.y;

    const int THREADS_PER_ROW = TILE_M / (BLOCK_SIZE / TILE_N);
    const int THREADS_PER_COL = TILE_N / (BLOCK_SIZE / TILE_M);
    
    // Calculate which element(s) this thread computes
    int row_in_tile = (thread_id / THREADS_PER_COL) % TILE_M;
    int col_in_tile = (thread_id % THREADS_PER_COL) * (TILE_N / THREADS_PER_COL);
    
    // Initialize accumulators
    float accum[TILE_N / THREADS_PER_COL];
    for (int i = 0; i < TILE_N / THREADS_PER_COL; i++) {
        accum[i] = 0.0f;
    }
    
    // Loop over K dimension in tiles
    for (int k_tile = 0; k_tile < K; k_tile += TILE_K) {
        // Load A tile into shared memory (coalesced access)
        for (int load_idx = thread_id; load_idx < TILE_M * TILE_K; load_idx += BLOCK_SIZE) {
            int row = load_idx / TILE_K;
            int col = load_idx % TILE_K;
            int global_row = block_row * TILE_M + row;
            int global_col = k_tile + col;
            
            if (global_row < M && global_col < K) {
                As[row][col] = A[global_row * K + global_col];
            } else {
                As[row][col] = 0.0f;
            }
        }
        
        // Load B tile into shared memory (coalesced access)
        for (int load_idx = thread_id; load_idx < TILE_K * TILE_N; load_idx += BLOCK_SIZE) {
            int row = load_idx / TILE_N;
            int col = load_idx % TILE_N;

However, adding many 1e-8 values together first (e.g., 8 of them yield 8e-8) still cannot affect 2000.0, so the final sum is always 2000.0? Actually, 2000 + 8e-8 = 2000.0 in float32 because the difference is below the least significant bit. To create a difference, use larger tiny values that can accumulate into a value capable of altering the sum when added at the right moment. For instance, set the first half to 1.0 and the second half to 1e-4. The sum of the tiny part is ~949*1e-4=0.0949, which is representable. The order in which the tiny additions are interleaved with the large ones determines whether they are fully added or partially lost. By varying the grouping, the final sums differ.
