# Define the custom CUDA kernel for ReLU
relu_source = """
#include <torch/extension.h>

template<typename scalar_t>
__global__ void relu_kernel(const scalar_t* __restrict__ input, 
                           scalar_t* __restrict__ output, 
                           int64_t size) {
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx < size) {
        scalar_t val = input[idx];
        output[idx] = val > static_cast<scalar_t>(0) ? val : static_cast<scalar_t>(0);
    }
}

torch::Tensor relu_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t size = input.numel();
    
    const int block_size = 256;
    const int grid_size = (size + block_size - 1) / block_size;
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "relu_cuda", ([&] {
        relu_kernel<scalar_t><<<grid_size, block_size>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            size
        );
    }));
    
    return output;
}
"""
