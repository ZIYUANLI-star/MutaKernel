#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

template<typename scalar_t>
__global__ void avg_pool_3d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int input_depth,
    const int input_height,
    const int input_width,
    const int output_depth,
    const int output_height,
    const int output_width,
    const int kernel_d,
    const int kernel_h,
    const int kernel_w,
    const int stride_d,
    const int stride_h,
    const int stride_w,
    const int pad_d,
    const int pad_h,
    const int pad_w,
    const bool count_include_pad
) {
    // Use 2D thread blocks for better memory coalescing
    const int tid = threadIdx.x;
    const int bid = blockIdx.x;
    
    // Each block processes multiple output positions
    const int elements_per_block = 256;  // Number of output positions per block
    const int start_idx = bid * elements_per_block;
    
    // Process multiple output positions per thread
    for (int linear_idx = start_idx + tid; linear_idx < start_idx + elements_per_block; linear_idx += blockDim.x) {
        if (linear_idx >= batch_size * channels * output_depth * output_height * output_width) {
            break;
        }
        
        // Decompose linear index
        int idx = linear_idx;
        const int ow = idx % output_width;
        idx /= output_width;
        const int oh = idx % output_height;
        idx /= output_height;
        const int od = idx % output_depth;
        idx /= output_depth;
        const int c = idx % channels;
        idx /= channels;
        const int b = idx;
        
        // Calculate input window start positions
        const int id_start = od * stride_d - pad_d;
        const int ih_start = oh * stride_h - pad_h;
        const int iw_start = ow * stride_w - pad_w;
        
        // Calculate clamped window bounds
        const int id_end = min(id_start + kernel_d, input_depth);
        const int ih_end = min(ih_start + kernel_h, input_height);
        const int iw_end = min(iw_start + kernel_w, input_width);
        const int id_start_clamped = max(id_start, 0);
        const int ih_start_clamped = max(ih_start, 0);
        const int iw_start_clamped = max(iw_start, 0);
        
        // Pre-calculate base index for this batch/channel
        const int base_input_idx = ((b * channels + c) * input_depth) * input_height * input_width;
        
        // Accumulate sum
        scalar_t sum = scalar_t(0);
        int count = 0;
        
        // Optimized loop with reduced index calculations
        for (int id = id_start_clamped; id < id_end; ++id) {
            const int depth_offset = id * input_height * input_width;
            for (int ih = ih_start_clamped; ih < ih_end; ++ih) {
                const int height_offset = ih * input_width;
                const int row_start_idx = base_input_idx + depth_offset + height_offset;
                
                // Process contiguous width elements
                for (int iw = iw_start_clamped; iw < iw_end; ++iw) {
                    sum += input[row_start_idx + iw];
                    count++;
                }
            }
        }
        
        // Calculate average
        scalar_t avg;
        if (count_include_pad) {
            avg = sum / scalar_t(kernel_d * kernel_h * kernel_w);
        } else {
            avg = (count > 0) ? sum / scalar_t(count) : scalar_t(0);
        }
        
        // Write output
        const int output_idx = ((b * channels + c) * output_depth + od) * output_height * output_width
                             + oh * output_width + ow;
        output[output_idx] = avg;
    }
}
