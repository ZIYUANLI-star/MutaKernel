import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define optimized CUDA kernel for 1D average pooling
avg_pool1d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

template<typename scalar_t>
__global__ void avg_pool1d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int input_length,
    const int output_length,
    const int kernel_size,
    const int stride,
    const int padding,
    const bool count_include_pad
) {
    // Each thread block processes multiple output positions for better memory coalescing
    const int batch_channel_idx = blockIdx.x;
    const int batch_idx = batch_channel_idx / channels;
    const int channel_idx = batch_channel_idx % channels;
    
    // Each thread processes one output position in the length dimension
    const int output_pos = threadIdx.x + blockIdx.y * blockDim.x;
    
    if (output_pos >= output_length) return;
    
    // Calculate input start position
    const int input_start = output_pos * stride - padding;
    const int input_end = input_start + kernel_size;
    
    // Get base input index for this batch and channel
    const int base_input_idx = (batch_idx * channels + channel_idx) * input_length;
    
    // Accumulate sum
    scalar_t sum = static_cast<scalar_t>(0.0);
    int count = 0;
    
    // Handle boundaries
    const int start_idx = max(0, input_start);
    const int end_idx = min(input_length, input_end);
    
    // Process elements
    for (int k = start_idx; k < end_idx; ++k) {
        sum += input[base_input_idx + k];
        count++;
    }
    
    // Handle padding if count_include_pad is true
    if (count_include_pad) {
        count = kernel_size;
        // Recalculate sum including padding zeros
        sum = static_cast<scalar_t>(0.0);
        for (int k = input_start; k < input_end; k++) {
            if (k >= 0 && k < input_length) {
                sum += input[base_input_idx + k];
            }
        }
    }
    
    // Write output
    if (count > 0) {
        const int output_idx = (batch_idx * channels + channel_idx) * output_length + output_pos;
        output[output_idx] = sum / static_cast<scalar_t>(count);
    }
}

torch::Tensor avg_pool1d_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    bool count_include_pad
) {
    // Check input dimensions
    TORCH_CHECK(input.dim() == 3, "Input must be 3D (batch, channels, length)");
    TORCH_CHECK(input.is_cuda(), "Input must be a CUDA tensor");
    
    // Input dimensions
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int input_length = input.size(2);
    
    // Calculate output dimensions
    const int output_length = ((input_length + 2 * padding - kernel_size) / stride) + 1;
    
    // Create output tensor
    auto output = torch::empty({batch_size, channels, output_length}, 
                              input.options());
    
    // Optimized kernel configuration
    const int threads_per_block = 256;
    const int blocks_x = batch_size * channels;
    const int blocks_y = (output_length + threads_per_block - 1) / threads_per_block;
    
    dim3 grid_dim(blocks_x, blocks_y);
    dim3 block_dim(threads_per_block);
    
    // Launch kernel
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(
        input.scalar_type(), "avg_pool1d_cuda", ([&] {
            avg_pool1d_kernel<scalar_t><<<grid_dim, block_dim>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size,
                channels,
                input_length,
                output_length,
                kernel_size,
                stride,
                padding,
                count_include_pad
            );
        })
    );
    
    return output;
}
"""
