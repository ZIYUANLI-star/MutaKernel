__global__ void relu_kernel(const scalar_t* __restrict__ input,

    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;

        output[idx] = val > static_cast<scalar_t>(0) ? val : static_cast<scalar_t>(0);
