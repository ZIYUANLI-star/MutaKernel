template<typename scalar_t>
__global__ void maxpool1d_forward_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int64_t* __restrict__ indices,
    const int batch_size,
    const int channels,
    const int input_length,
    const int output_length,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation,
    const bool return_indices
) {

    const int channel = blockIdx.y;
    const int output_pos = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (output_pos >= output_length) return;
    
    const int input_start = output_pos * stride - padding;
    scalar_t max_val = -INFINITY;
    int max_idx = -1;

    #pragma unroll(4)
    for (int k = 0; k < kernel_size; k++) {
        const int input_pos = input_start + k * dilation;
        
        if (input_pos >= 0 && input_pos < input_length) {
            const int input_idx = ((batch * channels + channel) * input_length + input_pos);
            const scalar_t val = input[input_idx];
            
            if (val > max_val) {
                max_val = val;
                max_idx = input_pos;
            }
        }
    }
    
    const int output_idx = ((batch * channels + channel) * output_length + output_pos);
    output[output_idx] = max_val;
    
    if (return_indices && indices != nullptr) {
        indices[output_idx] = max_idx;
    }
}
