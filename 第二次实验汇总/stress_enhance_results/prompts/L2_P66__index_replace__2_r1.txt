#include <torch/extension.h>
#include <cuda_runtime.h>
#include <curand_kernel.h>
#include <math.h>

__global__ void fused_matmul_dropout_mean_softmax_kernel(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int batch_size,
    int in_features,
    int out_features,
    float dropout_p,
    float dropout_scale,
    unsigned long long seed,
    bool training
) {
    int batch_idx = blockIdx.x;
    int out_idx = threadIdx.x;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    // Initialize random state
    curandStatePhilox4_32_10_t state;
    curand_init(seed, batch_idx * out_features + out_idx, 0, &state);
    
    // Perform matmul for this output feature
    float sum = 0.0f;
    const float* input_row = input + batch_idx * in_features;
    const float* weight_col = weight + out_idx * in_features;
    
    #pragma unroll 4
    for (int i = 0; i < in_features; i++) {
        sum += input_row[i] * weight_col[i];
    }
    
    // Add bias
    if (bias != nullptr) {
        sum += bias[out_idx];
    }
    
    // Apply dropout during training
    if (training && dropout_p > 0.0f) {
        float random_val = curand_uniform(&state);
        if (random_val < dropout_p) {
            sum = 0.0f;
        } else {
            sum *= dropout_scale;
        }
    }
    
    // Store intermediate result in shared memory for mean calculation
    extern __shared__ float shared_sum[];
    shared_sum[out_idx] = sum;
    __syncthreads();
    
    // Calculate mean across output features (reduction)
    if (out_idx == 0) {
        float mean_val = 0.0f;
        for (int i = 0; i < out_features; i++) {
            mean_val += shared_sum[i];
        }
        mean_val /= out_features;
        
        // Apply softmax (since we have only one value after mean, softmax becomes exp(mean)/exp(mean) = 1.0)
        // Actually, softmax over a single element is always 1.0
        output[batch_idx] = 1.0f;
    }
}

torch::Tensor fused_matmul_dropout_mean_softmax_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float dropout_p,
    bool training
) {
    auto batch_size = input.size(0);
    auto in_features = input.size(1);
    auto out_features = weight.size(0);

    // Calculate dropout scale factor

    // Generate random seed
    unsigned long long seed = time(NULL);
    
    // Launch kernel with optimized configuration
    dim3 blocks(batch_size);
    int threads_count = (out_features < 1024) ? out_features : 1024;

        batch_size,
        in_features,
        out_features,
        dropout_p,
        dropout_scale,
        seed,
        training
    );
    
    return output;
}
