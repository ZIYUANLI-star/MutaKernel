# Define optimized CUDA kernel for upper triangular matrix multiplication
triu_matmul_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void triu_matmul_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int N
) {
    // Each thread block computes a tile of the output matrix
    const int TILE_SIZE = 32;
    const int ROW_TILES = (N + TILE_SIZE - 1) / TILE_SIZE;
    
    // Shared memory for tiles of A and B
    __shared__ float As[TILE_SIZE][TILE_SIZE];
    __shared__ float Bs[TILE_SIZE][TILE_SIZE];
    
    // Thread indices within block
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Block indices
    int blockRow = blockIdx.y;
    int blockCol = blockIdx.x;
    
    // Only compute upper triangular blocks
    if (blockCol * TILE_SIZE < blockRow * TILE_SIZE) {
        return;
    }
    
    // Initialize accumulator
    float sum = 0.0f;
    
    // Loop over tiles
    for (int tile = blockRow; tile <= blockCol; tile++) {
        // Load A tile
        int A_row = blockRow * TILE_SIZE + ty;
        int A_col = tile * TILE_SIZE + tx;
        if (A_row < N && A_col < N && A_col >= A_row) {
            As[ty][tx] = A[A_row * N + A_col];
        } else {
            As[ty][tx] = 0.0f;
        }
        
        // Load B tile
        int B_row = tile * TILE_SIZE + ty;
        int B_col = blockCol * TILE_SIZE + tx;
        if (B_row < N && B_col < N && B_col >= B_row) {
            Bs[ty][tx] = B[B_row * N + B_col];
        } else {
            Bs[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial product for this tile
        for (int k = 0; k < TILE_SIZE; k++) {
            sum += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    // Write result
    int row = blockRow * TILE_SIZE + ty;
    int col = blockCol * TILE_SIZE + tx;
    
    if (row < N && col < N) {
        if (col >= row) {
            C[row * N + col] = sum;
        } else {
            C[row * N + col] = 0.0f;
        }
    }
}

torch::Tensor triu_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    auto C = torch::zeros({N, N}, A.options());
    
    // Use 32x32 thread blocks for better memory coalescing
    const int TILE_SIZE = 32;
    dim3 block_size(TILE_SIZE, TILE_SIZE);
    
    // Only launch blocks for upper triangular region
    int grid_x = (N + TILE_SIZE - 1) / TILE_SIZE;
    int grid_y = (N + TILE_SIZE - 1) / TILE_SIZE;
    dim3 grid_size(grid_x, grid_y);
    
    // Set shared memory size
    size_t shared_mem_size = 2 * TILE_SIZE * TILE_SIZE * sizeof(float);
    
    triu_matmul_kernel<<<grid_size, block_size, shared_mem_size>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        N
    );
    
    return C;
}
"""
