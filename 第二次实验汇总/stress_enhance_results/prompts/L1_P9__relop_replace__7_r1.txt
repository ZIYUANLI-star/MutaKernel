#include <cuda_runtime.h>

// Optimized kernel for A(M,K) * B(K,N) where M=16384, K=16, N=16384
// Using shared memory tiling and register blocking
__global__ void tall_skinny_matmul_kernel(
    const float* __restrict__ A,  // shape: [M, K] where K=16
    const float* __restrict__ B,  // shape: [K, N] where K=16
    float* __restrict__ C,        // shape: [M, N]
    int M, int K, int N
) {
    // Tile sizes
    const int TILE_M = 64;
    const int TILE_N = 64;
    
    // Block dimensions
    const int BLOCK_SIZE = 256;
    const int THREADS_PER_ROW = TILE_M / 4;  // 16 threads handle one row
    const int THREADS_PER_COL = TILE_N / 4;  // 16 threads handle one column
    
    // Shared memory for A and B tiles
    __shared__ float A_tile[TILE_M][16];
    __shared__ float B_tile[16][TILE_N];
    
    // Thread indices
    int tid = threadIdx.x;
    int block_row = blockIdx.y * TILE_M;
    int block_col = blockIdx.x * TILE_N;
    
    // Each thread handles 4x4 elements
    int thread_row = (tid / THREADS_PER_COL) * 4;
    int thread_col = (tid % THREADS_PER_COL) * 4;
    
    // Global row/col indices for this thread
    int global_row = block_row + thread_row;
    int global_col = block_col + thread_col;
    
    // Register accumulators for 4x4 output tile
    float accum[4][4] = {{0.0f}};
    
    // Loop over K dimension in chunks of 16
    for (int k_step = 0; k_step < K; k_step += 16) {
        // Load A tile into shared memory (coalesced)
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            int load_row = block_row + thread_row + i;
            int load_k = k_step + (tid % 16);
            
            if (load_row < M && load_k < K) {
                A_tile[thread_row + i][tid % 16] = A[load_row * K + load_k];
            } else if (load_k < K) {
                A_tile[thread_row + i][tid % 16] = 0.0f;
            }
        }
        
        // Load B tile into shared memory (coalesced)
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            int load_col = block_col + thread_col + j;
            int load_k = k_step + (tid / 16);
            
            if (load_k < K && load_col < N) {
                B_tile[tid / 16][thread_col + j] = B[load_k * N + load_col];
            } else if (load_k < K) {
                B_tile[tid / 16][thread_col + j] = 0.0f;
            }
        }
        
        __syncthreads();
        
        // Compute partial matrix multiplication
        #pragma unroll
        for (int k = 0; k < 16; k++) {
            // Load A values for this thread
            float a_vals[4];
            #pragma unroll
            for (int i = 0; i < 4; i++) {
                a_vals[i] = A_tile[thread_row + i][k];
            }
            
            // Load B values for this thread
            float b_vals[4];
            #pragma unroll
            for (int j = 0; j < 4; j++) {
                b_vals[j] = B_tile[k][thread_col + j];
            }
            
            // Update accumulators
            #pragma unroll
            for (int i = 0; i < 4; i++) {
                #pragma unroll
                for (int j = 0; j < 4; j++) {
                    accum[i][j] += a_vals[i] * b_vals[j];
                }
            }
        }
        
        __syncthreads();
    }
    
    // Write results to global memory
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        #pragma unroll

            int row = global_row + i;
            int col = global_col + j;

                C[row * N + col] = accum[i][j];
            }
        }
    }
}
