import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for vectorized sigmoid
sigmoid_kernel_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAStream.h>

// Vectorized sigmoid using 1/(1+exp(-x))
__global__ void vectorized_sigmoid_kernel(const float* __restrict__ input, 
                                          float* __restrict__ output, 
                                          int size) {
    const int idx = blockIdx.x * blockDim.x * 4 + threadIdx.x * 4;
    
    if (idx + 3 < size) {
        // Load 4 elements at once
        float4 in_vec;
        in_vec.x = input[idx];
        in_vec.y = input[idx + 1];
        in_vec.z = input[idx + 2];
        in_vec.w = input[idx + 3];
        
        // Compute sigmoid for all 4 elements
        float4 out_vec;
        out_vec.x = 1.0f / (1.0f + expf(-in_vec.x));
        out_vec.y = 1.0f / (1.0f + expf(-in_vec.y));
        out_vec.z = 1.0f / (1.0f + expf(-in_vec.z));
        out_vec.w = 1.0f / (1.0f + expf(-in_vec.w));
        
        // Store 4 elements at once
        reinterpret_cast<float4*>(output)[blockIdx.x * blockDim.x + threadIdx.x] = out_vec;
    } else {
        // Handle remaining elements
        for (int i = 0; i < 4; i++) {
            if (idx + i < size) {
                float x = input[idx + i];
                output[idx + i] = 1.0f / (1.0f + expf(-x));
            }
        }
    }
}

torch::Tensor vectorized_sigmoid_cuda(torch::Tensor input) {
    auto size = input.numel();
    auto output = torch::empty_like(input);
    
    const int block_size = 256;  // 256 threads per block
    const int vectorized_size = size / 4;
    const int num_blocks = (vectorized_size + block_size - 1) / block_size;
    
    cudaStream_t stream = c10::cuda::getCurrentCUDAStream();
    
    if (input.scalar_type() == torch::kFloat32) {
        vectorized_sigmoid_kernel<<<num_blocks, block_size, 0, stream>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            size
        );
    } else {
        // Fallback to PyTorch for other types
        output = torch::sigmoid(input);
    }
    
    return output;
}
"""
