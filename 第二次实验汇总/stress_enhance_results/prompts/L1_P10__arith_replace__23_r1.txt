import torch

# Define optimized CUDA kernel for batched matrix multiplication using shared memory
batched_matmul_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BLOCK_SIZE 16

__global__ void batched_matmul_kernel(
    const float* __restrict__ A,   // [N, M, K]
    const float* __restrict__ B,   // [K, L]
    float* __restrict__ C,         // [N, M, L]
    int N, int M, int K, int L
) {
    // Shared memory for tile of A and B
    __shared__ float As[BLOCK_SIZE][BLOCK_SIZE];
    __shared__ float Bs[BLOCK_SIZE][BLOCK_SIZE];
    
    // Thread indices within block
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Block indices
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;  // batch dimension
    
    if (bz >= N) return;
    
    // Output position
    int row = bx * BLOCK_SIZE + tx;
    int col = by * BLOCK_SIZE + ty;
    
    // Accumulator for this thread
    float sum = 0.0f;
    
    // Loop over tiles
    for (int t = 0; t < K; t += BLOCK_SIZE) {
        // Load tile from A into shared memory
        int A_row = row;
        int A_col = t + ty;
        if (A_row < M && A_col < K) {
            As[tx][ty] = A[bz * M * K + A_row * K + A_col];
        } else {
            As[tx][ty] = 0.0f;
        }
        
        // Load tile from B into shared memory
        int B_row = t + tx;
        int B_col = col;
        if (B_row < K && B_col < L) {
            Bs[tx][ty] = B[B_row * L + B_col];
        } else {
            Bs[tx][ty] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial product
        for (int k = 0; k < BLOCK_SIZE; ++k) {
            sum += As[tx][k] * Bs[k][ty];
        }
        
        __syncthreads();
    }
    
    // Write result to C
    if (row < M && col < L) {
        C[bz * M * L + row * L + col] = sum;
    }
}

torch::Tensor batched_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    // Get dimensions
    int N = A.size(0);
    int M = A.size(1);
    int K = A.size(2);
    int L = B.size(1);
    
    // Create output tensor
    auto C = torch::zeros({N, M, L}, A.options());
    
    // Optimized block and grid configuration
    dim3 block(BLOCK_SIZE, BLOCK_SIZE);
    dim3 grid((M + BLOCK_SIZE - 1) / BLOCK_SIZE, 
              (L + BLOCK_SIZE - 1) / BLOCK_SIZE,
              N);  // One block per batch
    
    // Launch kernel
    batched_matmul_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        N, M, K, L
    );
    
    // Check for CUDA errors
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(cudaGetErrorString(err));
    }
    
    return C;
}
"""

batched_matmul_cpp_source = "torch::Tensor batched_matmul_cuda(torch::Tensor A, torch::Tensor B);"

# Compile the inline CUDA code with optimization flags
batched_matmul = load_inline(

    functions=["batched_matmul_cuda"],
    verbose=False,
    extra_cflags=["-O3", "-DNDEBUG"],
    extra_ldflags=[],
    with_cuda=True,
)

class ModelNew(nn.Module):

    Optimized 3D tensor-matrix multiplication with custom CUDA kernel using shared memory.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.batched_matmul = batched_matmul
    
    def forward(self, A, B):
        """
        Performs optimized 3D tensor-matrix multiplication.

        Args:
            A (torch.Tensor): Input 3D tensor of shape (N, M, K).
            B (torch.Tensor): Input matrix of shape (K, L).

        Returns:
            torch.Tensor: Output tensor of shape (N, M, L).
        """
        # Ensure tensors are contiguous and on CUDA
        A = A.contiguous().cuda()
        B = B.contiguous().cuda()
        
        return self.batched_matmul.batched_matmul_cuda(A, B)
