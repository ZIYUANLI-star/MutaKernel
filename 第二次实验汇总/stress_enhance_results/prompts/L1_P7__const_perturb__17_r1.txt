import torch

# Define custom CUDA kernel for optimized matrix multiplication with small K
matmul_small_k_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

template<typename T>
__global__ void matmul_small_k_kernel(
    const T* __restrict__ A,
    const T* __restrict__ B,
    T* __restrict__ C,
    int M, int N, int K) {
    
    // Tile-based approach with shared memory
    __shared__ T A_tile[32][32];
    __shared__ T B_tile[32][32];
    
    int row = blockIdx.y * 32 + threadIdx.y;
    int col = blockIdx.x * 32 + threadIdx.x;
    
    T sum = static_cast<T>(0.0);
    
    // Loop over tiles
    for (int tile = 0; tile < (K + 31) / 32; tile++) {
        int k_offset = tile * 32;
        
        // Load A tile
        int a_row = row;
        int a_col = k_offset + threadIdx.x;
        if (a_row < M && a_col < K) {
            A_tile[threadIdx.y][threadIdx.x] = A[a_row * K + a_col];
        } else {
            A_tile[threadIdx.y][threadIdx.x] = static_cast<T>(0.0);
        }
        
        // Load B tile
        int b_row = k_offset + threadIdx.y;
        int b_col = col;
        if (b_row < K && b_col < N) {
            B_tile[threadIdx.y][threadIdx.x] = B[b_row * N + b_col];
        } else {
            B_tile[threadIdx.y][threadIdx.x] = static_cast<T>(0.0);
        }
        
        __syncthreads();
        
        // Compute partial sum for this tile
        for (int k = 0; k < 32; k++) {
            sum += A_tile[threadIdx.y][k] * B_tile[k][threadIdx.x];
        }
        
        __syncthreads();
    }
    
    // Write result
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

torch::Tensor matmul_small_k_cuda(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    // Use 32x32 thread blocks for better memory coalescing
    dim3 block_dim(32, 32);
    dim3 grid_dim(
        (N + 31) / 32,
        (M + 31) / 32
    );
    
    AT_DISPATCH_FLOATING_TYPES(A.scalar_type(), "matmul_small_k", [&]() {
        matmul_small_k_kernel<scalar_t><<<grid_dim, block_dim>>>(
            A.data_ptr<scalar_t>(),
            B.data_ptr<scalar_t>(),
            C.data_ptr<scalar_t>(),
            M, N, K);
    });
    
    return C;
}
"""
